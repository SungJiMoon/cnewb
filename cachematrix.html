## Following function cache the inverse of matrix inputed. ## The base function of "solve" is used to get inverse of the matrix makeCacheMatrix <- function(x = matrix()) { m <- NULL set <- function(y) { x <<- y m <<- NULL } get <- function() x setsolve <- function(solve) m <<- solve getsolve <- function() m list(set = set, get = get, setsolve = setsolve, getsolve = getsolve) } ## Following function return the inverse of the given result from makeCacheMatrix function ## If the inverse of matrix was already calcuated, it will get the calcuated inverse skipping the inverse calculation. cacheSolve <- function(x, ...) { m <- x$getsolve() if(!is.null(m)) { message("getting cached data") return(m) } data <- x$get() m <- solve(data, ...) x$setsolve(m) m }
